\begin{lstlisting}[style=pythonstyle, caption=Performance Analysis Code]
def analyze_metrics(data: dict) -> dict:
    """Analyze business metrics."""
    results = {
        'revenue_growth': (data['q1'] - data['q4']) / data['q4'],
        'profit_margin': data['profit'] / data['revenue'],
        'trend': 'positive' if data['q1'] > data['q4'] else 'negative'
    }
    return results
\end{lstlisting}
